# Funcin de limpieza para topic modeling
def text_clean_for_topic_lemma(text: str):

    # Convertir a minscula
    txt = text.lower()

    # Remover lista negras de frases
    for black_text in black_text_list:
        txt = txt.replace(black_text, " ")

    # Remover signos de puntuacin
    txt = re.sub('[%s]' % re.escape(string.punctuation), ' ', txt)

    # Remover nmeros
    txt = re.sub(r'\w*\d+\w*', '', txt)

    # Remover caracteres unicode y nmeros excepto tldes y 
    txt = re.sub("[^a-z ]+", "", txt)

    # Aplicar Lemmatize
    txt = " ".join([word.lemma_ for word in lm(txt) if word.lemma_ not in stop_words])

    # Remover palabras menor igual que tres caracteres
    txt = re.sub(r'\b\w{1,3}\b', '', txt)

    # Remover espacios vacios
    txt = txt = re.sub(r"\s+", " ", txt)

    # Remover espacios en inicio y fin
    txt = txt.strip()

    return txt

# Funcin de limpieza para sumarizacin
# No incluye lematizacin
# Limpia caracteres menores iguales que tres
def text_clean_for_topic(text: str):

    # Convertir a minscula
    txt = text.lower()

    # Remover lista negras de frases
    for black_text in black_text_list:
        txt = txt.replace(black_text, " ")

    # Remover signos de puntuacin
    txt = re.sub('[%s]' % re.escape(string.punctuation), ' ', txt)

    # Remover nmeros
    txt = re.sub(r'\w*\d+\w*', '', txt)

    # Remover caracteres unicode y nmeros excepto tldes y 
    txt = re.sub("[^a-z ]+", "", txt)

    # Remover stop_words
    txt = " ".join([w for w in txt.split() if w not in stop_words])

    # Remover palabras menor igual que tres caracteres
    txt = re.sub(r'\b\w{1,3}\b', '', txt)

    # Remover espacios vacios
    txt = txt = re.sub(r"\s+", " ", txt)

    # Remover espacios en inicio y fin
    txt = txt.strip()

    return txt

# Funcin de limpieza para sumarizacin
# No incluye lematizacin
# No incluye stop words
# Limpia caracteres menores iguales que uno
def text_clean_for_summarize(text: str):

    # Convertir a minscula
    txt = text.lower()

    # Remover lista negras de frases
    for black_text in black_text_list:
        txt = txt.replace(black_text, " ")

    # Remover signos de puntuacin
    txt = re.sub('[%s]' % re.escape(string.punctuation), ' ', txt)

    # Remover nmeros
    txt = re.sub(r'\w*\d+\w*', '', txt)

    # Remover caracteres unicode y nmeros excepto tldes y 
    txt = re.sub("[^a-z ]+", "", txt)

    # Remover palabras menor igual que un carater
    txt = re.sub(r'\b\w{1,1}\b', '', txt)

    # Remover espacios vacios
    txt = txt = re.sub(r"\s+", " ", txt)

    # Remover espacios en inicio y fin
    txt = txt.strip()

    return txt
